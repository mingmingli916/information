\title{à de en}
\date{}


\section{En général}


\begin{description}
\item[saveur/sorte] à + déteminant + nom
\item[usage] à verbe/nom
\item[matière] de + nom
\item[origin] de + lieu
\item[traitement] en + nom
\item[traitement] adjectif
\end{description}


\section{Le Exemple}


\begin{itemize}
\item boeuf haché
\item crème à café
\item fromage à la crème
\item salade de macaroni
\item pomme de terre
\end{itemize}
